=========================================== = 3D FPS SHOOTER – UNITY = = Complete Scripts + Setup Guide = =========================================== ------------------------------------------- 1. PROJECT STRUCTURE ------------------------------------------- Assets/ ├── Scripts/ │ ├── PlayerMovement.cs │ ├── MouseLook.cs │ ├── Gun.cs │ └── EnemyAI.cs ├── Prefabs/ │ ├── Player.prefab │ └── Enemy.prefab ├── Scenes/ │ └── MainScene.unity └── Materials/ ------------------------------------------- 2. PLAYER MOVEMENT SCRIPT (WASD + Jump) ------------------------------------------- File: PlayerMovement.cs using UnityEngine; public class PlayerMovement : MonoBehaviour { public float speed = 6f; public float gravity = -9.81f; public float jumpHeight = 2f; CharacterController controller; Vector3 velocity; void Start() { controller = GetComponent(); } void Update() { float x = Input.GetAxis("Horizontal"); float z = Input.GetAxis("Vertical"); Vector3 move = transform.right * x + transform.forward * z; controller.Move(move * speed * Time.deltaTime); if (Input.GetButtonDown("Jump") && controller.isGrounded) { velocity.y = Mathf.Sqrt(jumpHeight * -2f * gravity); } velocity.y += gravity * Time.deltaTime; controller.Move(velocity * Time.deltaTime); } } ------------------------------------------- 3. MOUSE LOOK SCRIPT ------------------------------------------- File: MouseLook.cs using UnityEngine; public class MouseLook : MonoBehaviour { public float sensitivity = 300f; public Transform playerBody; float xRotation = 0f; void Start() { Cursor.lockState = CursorLockMode.Locked; } void Update() { float mouseX = Input.GetAxis("Mouse X") * sensitivity * Time.deltaTime; float mouseY = Input.GetAxis("Mouse Y") * sensitivity * Time.deltaTime; xRotation -= mouseY; xRotation = Mathf.Clamp(xRotation, -70f, 70f); transform.localRotation = Quaternion.Euler(xRotation, 0f, 0f); playerBody.Rotate(Vector3.up * mouseX); } } ------------------------------------------- 4. GUN + SHOOTING SCRIPT ------------------------------------------- File: Gun.cs using UnityEngine; public class Gun : MonoBehaviour { public float damage = 25f; public float range = 100f; public Camera fpsCam; public ParticleSystem muzzleFlash; void Update() { if (Input.GetButtonDown("Fire1")) { Shoot(); } } void Shoot() { muzzleFlash.Play(); RaycastHit hit; if (Physics.Raycast(fpsCam.transform.position, fpsCam.transform.forward, out hit, range)) { EnemyAI enemy = hit.transform.GetComponent(); if (enemy != null) { enemy.TakeDamage(damage); } } } } ------------------------------------------- 5. ENEMY AI SCRIPT ------------------------------------------- File: EnemyAI.cs using UnityEngine; using UnityEngine.AI; public class EnemyAI : MonoBehaviour { public float health = 50f; public NavMeshAgent agent; public Transform player; void Update() { agent.SetDestination(player.position); } public void TakeDamage(float amount) { health -= amount; if (health <= 0) Destroy(gameObject); } } ------------------------------------------- 6. HOW TO SET UP THE GAME IN UNITY ------------------------------------------- STEP 1 — Create a Unity 3D project. STEP 2 — Create folders: - Assets/Scripts - Assets/Prefabs - Assets/Scenes STEP 3 — Add a Scene: - File → New Scene - Save as "MainScene" STEP 4 — Create the Player: - Create an empty object "Player" - Add: • CharacterController component • PlayerMovement.cs - Add a Camera inside the Player: • Add MouseLook.cs on Camera • Drag Player -> playerBody field in inspector STEP 5 — Create the Gun: - Create empty object "Gun" under Camera - Add: • Gun.cs • Add a Particle System for muzzleFlash - Drag camera to fpsCam field STEP 6 — Create Enemy: - Create a 3D Capsule - Add: • NavMeshAgent • EnemyAI.cs - Drag Player to “player” field - Save as Enemy.prefab STEP 7 — Map: - Add a Terrain (GameObject → 3D → Terrain) - Add obstacles (Cubes, walls, buildings) STEP 8 — NavMesh: - Window → AI → Navigation - Bake the NavMesh so enemies can walk STEP 9 — Lighting & Graphics: - Add Directional Light - Set shadows to "Soft" - Add a Skybox (HDR) STEP 10 — Test the game: - Press Play - Move with WASD - Look around with Mouse - Shoot with Left Click ------------------------------------------- 7.
